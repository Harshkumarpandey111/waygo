Build WayGo — Full-Stack Travel Planner

A complete React.js frontend + Node.js/Express backend travel planner with cost calculator, transport modes, nearby places, and Google Maps navigation. Follow every step in order.

⚛️ React 18
🟢 Node.js + Express
🗺️ Google Maps API
⛽ Fuel Calculator
🚆 Train/Bus/Flight Fares
🏨 Hotels & Places
📍 Live Navigation

📐 What You'll Build

A 5-step travel planner that works like this:

StepComponentWhat it does
Step 1RouteFormEnter From/To, distance, travelers, dates
Step 2TransportPickerChoose Car/Bike (mileage calc) or Bus/Train/Flight/Cab (fares)
Step 3CostBreakdownItemised cost: fuel, toll, food, parking, misc
Step 4PlacesExplorerHotels, restaurants, petrol stations via Google Places API
Step 5JourneySummaryFull summary + launch Google Maps navigation
💡
This guide gives you every single file and every line of code. Just follow the steps in order: Prerequisites → Project Structure → Backend → Frontend → Run.

🖥️ backend/server.js

Create a new file called server.js inside the backend/ folder. Paste this entire code:

backend/server.js
const express = require('express');
const cors = require('cors');
const dotenv = require('dotenv');
const placesRouter = require('./routes/places');
const faresRouter = require('./routes/fares');
const fuelRouter = require('./routes/fuel');

dotenv.config();

const app = express();
const PORT = process.env.PORT || 5000;

// Middleware
app.use(cors({
  origin: ['http://localhost:5173', 'http://localhost:3000'],
  credentials: true
}));
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// Routes
app.use('/api/places', placesRouter);
app.use('/api/fares', faresRouter);
app.use('/api/fuel', fuelRouter);

// Health check
app.get('/', (req, res) => {
  res.json({
    message: '🚀 WayGo API is running!',
    version: '1.0.0',
    endpoints: [
      '/api/places/nearby',
      '/api/fares/estimate',
      '/api/fuel/calculate'
    ]
  });
});

// Error handler
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({ error: 'Something went wrong!' });
});

app.listen(PORT, () => {
  console.log(`✅ WayGo server running on http://localhost:${PORT}`);
});

📄 routes/fuel.js — Fuel cost calculator

backend/routes/fuel.js
const express = require('express');
const router = express.Router();

// POST /api/fuel/calculate
router.post('/calculate', (req, res) => {
  const {
    distance,        // km
    mileage,         // km/L
    fuelPrice,       // ₹/L
    tollCharges,     // ₹
    travelers,       // number
    vehicleType      // 'car' | 'bike'
  } = req.body;

  // Validate
  if (!distance || !mileage || !fuelPrice) {
    return res.status(400).json({ error: 'distance, mileage, fuelPrice are required' });
  }

  const litersNeeded = distance / mileage;
  const fuelCost = Math.round(litersNeeded * fuelPrice);
  const toll = tollCharges || 0;

  // Estimated stops (every 150km)
  const stops = Math.floor(distance / 150);
  const foodCost = Math.round(stops * travelers * 180);
  const parkingCost = Math.round(stops * 80);
  const miscCost = Math.round((fuelCost + toll) * 0.08);
  const totalCost = fuelCost + toll + foodCost + parkingCost + miscCost;

  res.json({
    success: true,
    breakdown: {
      litersNeeded: +litersNeeded.toFixed(1),
      fuelCost,
      tollCharges: toll,
      foodCost,
      parkingCost,
      miscCost,
      totalCost,
      perPersonCost: Math.round(totalCost / (travelers || 1))
    }
  });
});

module.exports = router;

const router = express.Router();

// Rate per km (₹) for each class
const FARE_RATES = {
  bus: {
    general:  { rate: 0.55, label: 'State Bus - General', icon: '🚌' },
    express:  { rate: 0.85, label: 'State Bus - Express', icon: '🚌' },
    volvo:    { rate: 1.40, label: 'Volvo AC Bus', icon: '🚌' }
  },
  train: {
    sl:   { rate: 0.50, label: 'Sleeper Class (SL)', icon: '🚆' },
    threeA: { rate: 1.60, label: 'AC 3-Tier (3A)', icon: '🚆' },
    twoA: { rate: 2.40, label: 'AC 2-Tier (2A)', icon: '🚆' },
    cc:   { rate: 1.30, label: 'AC Chair Car (CC)', icon: '🚆' }
  },
  flight: {
    economy:  { rate: 4.50, label: 'Economy Class', icon: '✈️' },
    business: { rate: 14.00, label: 'Business Class', icon: '✈️' }
  },
  cab: {
    hatchback: { rate: 12, label: 'Hatchback (Ola Mini)', icon: '🚕' },
    sedan:     { rate: 16, label: 'Sedan (Ola Prime)', icon: '🚕' },
    suv:       { rate: 22, label: 'SUV (Ola SUV)', icon: '🚕' }
  }
};

// GET /api/fares/estimate?type=train&distance=320&travelers=2
router.get('/estimate', (req, res) => {
  const { type, distance, travelers = 1 } = req.query;

  if (!FARE_RATES[type]) {
    return res.status(400).json({ error: 'Invalid transport type' });
  }

  const dist = parseFloat(distance);
  const trav = parseInt(travelers);
  const classes = FARE_RATES[type];

  const fares = Object.entries(classes).map(([key, data]) => ({
    key,
    label: data.label,
    icon: data.icon,
    perPersonFare: Math.round(dist * data.rate),
    totalFare: Math.round(dist * data.rate * trav),
    ratePerKm: data.rate
  }));

  res.json({ success: true, type, distance: dist, travelers: trav, fares });
});

module.exports = router;

📄 routes/places.js — Google Places proxy

backend/routes/places.js
const express = require('express');
const axios = require('axios');
const router = express.Router();

const GOOGLE_API_KEY = process.env.GOOGLE_MAPS_API_KEY;
const PLACES_BASE = 'https://maps.googleapis.com/maps/api/place';

// GET /api/places/nearby?location=28.6139,77.2090&type=hotel&radius=5000
router.get('/nearby', async (req, res) => {
  try {
    const { location, type = 'lodging', radius = 5000, keyword } = req.query;

    const params = {
      location, radius, type,
      key: GOOGLE_API_KEY,
      ...(keyword && { keyword })
    };

    const { data } = await axios.get(`${PLACES_BASE}/nearbysearch/json`, { params });

    const places = (data.results || []).slice(0, 8).map(p => ({
      id: p.place_id,
      name: p.name,
      address: p.vicinity,
      rating: p.rating || 0,
      ratingCount: p.user_ratings_total || 0,
      isOpen: p.opening_hours?.open_now,
      priceLevel: p.price_level,
      photo: p.photos?.[0]?.photo_reference
        ? `${PLACES_BASE}/photo?maxwidth=400&photoreference=${p.photos[0].photo_reference}&key=${GOOGLE_API_KEY}`
        : null,
      lat: p.geometry.location.lat,
      lng: p.geometry.location.lng,
      types: p.types
    }));

    res.json({ success: true, places, total: places.length });
  } catch (err) {
    console.error('Places error:', err.message);
    res.status(500).json({ error: 'Failed to fetch places', details: err.message });
  }
});

// GET /api/places/geocode?address=Delhi
router.get('/geocode', async (req, res) => {
  try {
    const { address } = req.query;
    const { data } = await axios.get(
      `https://maps.googleapis.com/maps/api/geocode/json`,
      { params: { address, key: GOOGLE_API_KEY } }
    );
    if (data.results?.[0]) {
      const { lat, lng } = data.results[0].geometry.location;
      res.json({ success: true, lat, lng, formattedAddress: data.results[0].formatted_address });
    } else {
      res.status(404).json({ error: 'Location not found' });
    }
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

module.exports = router;

🧠 context/TravelContext.jsx

This is the global state manager. All 5 steps read and write to this shared state.

src/context/TravelContext.jsx
import { createContext, useContext, useState } from 'react';

const TravelContext = createContext(null);

export function TravelProvider({ children }) {
  const [travel, setTravel] = useState({
    // Step 1
    origin: '',
    destination: '',
    distance: '',
    travelers: 1,
    travelDate: '',
    returnDate: '',
    // Step 2
    transport: '',
    mileage: '',
    fuelPrice: 96,
    tollCharges: 0,
    selectedFareClass: '',
    additionalCost: 0,
    // Step 3
    costBreakdown: null,
    totalCost: 0,
    // Step 4
    selectedPlaces: [],
    // Step 5
    destCoords: null
  });

  const update = (fields) =>
    setTravel(prev => ({ ...prev, ...fields }));

  const reset = () => setTravel({
    origin: '', destination: '', distance: '', travelers: 1,
    travelDate: '', returnDate: '', transport: '', mileage: '',
    fuelPrice: 96, tollCharges: 0, selectedFareClass: '',
    additionalCost: 0, costBreakdown: null, totalCost: 0,
    selectedPlaces: [], destCoords: null
  });

  return (
    <TravelContext.Provider value={{ travel, update, reset }}>
      {children}
    </TravelContext.Provider>
  );
}

  const [loading, setLoading] = useState(true);
  const isPersonal = ['car', 'bike'].includes(travel.transport);

  useEffect(() => {
    const fetchCosts = async () => {
      try {
        if (isPersonal) {
          const { data } = await axios.post(
            `${import.meta.env.VITE_API_URL}/fuel/calculate`,
            {
              distance: +travel.distance, mileage: +travel.mileage,
              fuelPrice: +travel.fuelPrice, tollCharges: +travel.tollCharges,
              travelers: +travel.travelers, vehicleType: travel.transport
            }
          );
          setBreakdown(data.breakdown);
          update({ costBreakdown: data.breakdown, totalCost: data.breakdown.totalCost });
        } else {
          // Public transport — compute from stored fareEstimate
          const fare = travel.fareEstimate || 0;
          const food = travel.travelers * 200;
          const extra = +travel.additionalCost || 0;

            transportFare: fare, foodCost: food,
            additionalCost: extra, totalCost: total,

export default function Step4_Places() {
  const navigate = useNavigate();
  const { travel, update } = useTravel();
  const [activeTab, setActiveTab] = useState('lodging');
  const [places, setPlaces] = useState({});
  const [loading, setLoading] = useState(false);
  const [coords, setCoords] = useState(null);

  // Geocode destination first
  useEffect(() => {
    if (!travel.destination) return;
    axios.get(`${import.meta.env.VITE_API_URL}/places/geocode?address=${encodeURIComponent(travel.destination)}`)
      .then(r => {
        const c = { lat: r.data.lat, lng: r.data.lng };
        setCoords(c);
        update({ destCoords: c });
      })
      .catch(() => {});
  }, []);

  // Fetch places for active tab
  useEffect(() => {
    if (!coords || places[activeTab]) return;
